--> 1. Seleziona gli utenti che hanno postato almeno un video <--


USE db_social

SELECT username 
FROM users u
JOIN medias m ON m.user_id = m.id;
WHERE m.`type` = 'video';



--> 2.Seleziona tutti i post senza Like (13) <--

SELECT *
FROM posts p 
LEFT JOIN likes l ON  p.id =  l.post_id 
WHERE l.post_id is null ;


--> 3. Conta il numero di like per ogni post (165) <--

SELECT l.post_id , count(l.post_id) as 'N째 Like'
FROM likes l 
group by l.post_id;



--> 4. Ordina gli utenti per il numero di media caricati (25)  <--

SELECT user_id, username, COUNT(*) as 'N째 media'
FROM users u 
JOIN medias m ON m.id = u.id 
GROUP BY u.id 
ORDRE BY u.username ;


--> 5. Ordina gli utenti per totale di likes ricevuti nei loro posts (25)  <--

SELECT u.id, u.username, count(*) as 'N째 totale di like'
FROM users u 
JOIN posts p ON u.id = p.user_id 
JOIN likes l ON p.id = l.post_id  
GROUP BY u.id 
ORDER BY 'N째 totale di like' ASC;
